### 🛡️ 模块三：风险扫描仪

#### 风险自检函数：
```python
风险自检函数(股票代码):
    风险项 = [
        {"检查": "是否为ST/*ST", "权重": 30, "建议": "❌ 坚决回避"},
        {"检查": "流通盘>50亿", "权重": 15, "建议": "⚠️ 谨慎参与"},
        {"检查": "近3日换手率<5%", "权重": 15, "建议": "⚠️ 关注活跃度"},
        {"检查": "有退市风险公告", "权重": 30, "建议": "❌ 坚决回避"},
        {"检查": "处于利好兑现阶段", "权重": 20, "建议": "⚠️ 谨慎追高"},
        {"检查": "近10日有连续跌停", "权重": 20, "建议": "⚠️ 关注流动性"}
    ]
    
    初始分 = 100
    扣分项 = []
    
    for 项 in 风险项:
        if 触发风险(项["检查"]):
            初始分 -= 项["权重"]
            扣分项.append(f"{项['建议']} {项['检查']}")
    
    风险分 = max(初始分, 0)
    
    if 风险分 >= 80:
        等级 = "🟢 低风险"
        操作 = "✅ 可参与"
    elif 风险分 >= 60:
        等级 = "🟡 中风险" 
        操作 = "⚠️ 谨慎参与"
    else:
        等级 = "🔴 高风险"
        操作 = "❌ 坚决回避"
    
    return {分数: 风险分, 等级: 等级, 扣分项: 扣分项, 建议: 操作}
```
